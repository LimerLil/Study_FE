Qua bài học trước, chúng ta đã được học về thành phần thẻ <head> là nơi chứa các thẻ meta

    <!DOCTYPE html>
        <html lang="en">
        <head>
            <meta charset="UTF-8">
            <meta name="viewport" content="width=device-width, initial-scale=1.0">
            <title>Midori3</title>
        </head>
     
    a { color: red}
    
    const fullName = "Nguyen Van A"
